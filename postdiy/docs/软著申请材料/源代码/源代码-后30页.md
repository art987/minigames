// ==================== 海报下载功能 ====================

/**
 * 下载海报到相册
 * 兼容WebView环境，通过JSBridge调用原生保存能力
 */
async function downloadPoster() {
    if (!checkBrandInfoComplete()) {
        showIncompleteInfoWarning();
        return;
    }

    showDownloadLoading();

    try {
        const posterElement = document.getElementById('posterCanvas');
        if (!posterElement) {
            throw new Error('海报容器未找到');
        }

        // 使用html2canvas生成海报
        const canvas = await html2canvas(posterElement, {
            useCORS: true,
            allowTaint: false,
            scale: 2,
            backgroundColor: '#ffffff',
            logging: false
        });

        const dataUrl = canvas.toDataURL('image/jpeg', 0.95);
        const fileName = generatePosterFileName();

        // WebView环境：通过JSBridge保存
        if (isInWebView()) {
            saveImageViaBridge(dataUrl, fileName);
        } else {
            // 浏览器环境：直接下载
            downloadImageInBrowser(dataUrl, fileName);
        }

        // 记录下载次数
        await recordDownloadUsage();

        showDownloadSuccess();

    } catch (error) {
        console.error('下载海报失败:', error);
        showDownloadError(error.message);
    } finally {
        hideDownloadLoading();
    }
}

/**
 * 检查品牌信息是否完整
 */
function checkBrandInfoComplete() {
    const brandName = document.getElementById('brandName').value.trim();
    const logoImg = document.querySelector('.brand-logo img');
    const qrCodeImg = document.querySelector('.qrcode img');

    if (!brandName) {
        return false;
    }

    if (!logoImg || !logoImg.src || logoImg.src.includes('placeholder')) {
        return false;
    }

    if (!qrCodeImg || !qrCodeImg.src || qrCodeImg.src.includes('placeholder')) {
        return false;
    }

    return true;
}

/**
 * 通过JSBridge保存图片
 */
function saveImageViaBridge(dataUrl, fileName) {
    // 多种WebView兼容方案
    const bridges = [
        () => window.webkit?.messageHandlers?.downloadImage?.postMessage({ dataUrl, fileName }),
        () => window.AndroidInterface?.saveImage?.(dataUrl, fileName),
        () => window.uni?.postMessage?.({ data: { action: 'downloadImage', dataUrl, fileName } }),
        () => window.uni?.webView?.postMessage?.({ data: { action: 'downloadImage', dataUrl, fileName } }),
        () => window.parent?.postMessage?.({ action: 'downloadImage', dataUrl, fileName }, '*')
    ];

    for (const bridge of bridges) {
        try {
            const result = bridge();
            if (result !== false) {
                console.log('通过JSBridge保存成功');
                return;
            }
        } catch (e) {
            console.warn('JSBridge调用失败:', e);
        }
    }

    // 所有Bridge都失败，回退到浏览器下载
    downloadImageInBrowser(dataUrl, fileName);
}

/**
 * 浏览器环境下载图片
 */
function downloadImageInBrowser(dataUrl, fileName) {
    const link = document.createElement('a');
    link.href = dataUrl;
    link.download = fileName;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
}

/**
 * 生成海报文件名
 */
function generatePosterFileName() {
    const brandName = document.getElementById('brandName').value.trim() || '海报';
    const timestamp = new Date().toISOString().slice(0, 19).replace(/[:-]/g, '');
    const randomSuffix = Math.random().toString(36).slice(2, 6);
    return `${brandName}_${timestamp}_${randomSuffix}.jpg`;
}

function initBrandInfoEditor() {
    // Logo上传
    const logoInput = document.getElementById('logoInput');
    const logoPreview = document.querySelector('.brand-logo img');

    logoInput.addEventListener('change', (e) => {
        const file = e.target.files[0];
        if (!file) return;

        if (!file.type.startsWith('image/')) {
            alert('请选择图片文件');
            return;
        }

        if (file.size > 5 * 1024 * 1024) {
            alert('图片大小不能超过5MB');
            return;
        }

        const reader = new FileReader();
        reader.onload = (event) => {
            logoPreview.src = event.target.result;
            saveBrandInfo();
        };
        reader.readAsDataURL(file);
    });

    // 商家名称输入
    const brandNameInput = document.getElementById('brandName');
    brandNameInput.addEventListener('input', debounce(saveBrandInfo, 500));

    // 二维码上传
    const qrCodeInput = document.getElementById('qrCodeInput');
    const qrCodePreview = document.querySelector('.qrcode img');

    qrCodeInput.addEventListener('change', (e) => {
        const file = e.target.files[0];
        if (!file) return;

        if (!file.type.startsWith('image/')) {
            alert('请选择图片文件');
            return;
        }

        if (file.size > 5 * 1024 * 1024) {
            alert('图片大小不能超过5MB');
            return;
        }

        const reader = new FileReader();
        reader.onload = (event) => {
            qrCodePreview.src = event.target.result;
            saveBrandInfo();
        };
        reader.readAsDataURL(file);
    });

    // 加载已保存的品牌信息
    loadSavedBrandInfo();
}

/**
 * 防抖函数
 */
function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        const later = () => {
            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}

/**
 * 验证促销文案

function validatePromoText(textarea, index) {
    const maxLength = [10, 15, 10]; // 三行文案的最大长度
    const value = textarea.value.trim();

    if (value.length > maxLength[index]) {
        textarea.style.borderColor = '#ff4444';
        showWarning(`第${index + 1}行文案不能超过${maxLength[index]}个字`);
    } else {
        textarea.style.borderColor = '#ddd';
    }
}

async function checkVipStatus() {
    const userId = localStorage.getItem('postdiy_user_id');

            return {
                isVip: result.data.isVip,
                vipExpireAt: result.data.vipExpireAt,
                downloadQuota: result.data.downloadQuota
            };
        }

    } catch (error) {
        console.error('检查VIP状态失败:', error);

    return { isVip: false, downloadQuota: 0 };

/**
 * 处理购买
 */
async function handlePurchase(packageId) {
    const userId = localStorage.getItem('postdiy_user_id');
    if (!userId) {
        showLoginModal();
        return;
    }

    try {
        const response = await fetch('https://api.peacelove.top/create-order', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ userId, packageId })
        });

        const result = await response.json();

        if (result.success && result.data.payUrl) {
            // WebView环境：打开独立收银台
            if (isInWebView()) {
                openPaymentPage(result.data.payUrl);
            } else {
                // 浏览器环境：直接跳转
                window.location.href = result.data.payUrl;
            }
        } else {
            alert('创建订单失败，请稍后重试');
        }

    } catch (error) {
        console.error('购买失败:', error);
        alert('购买失败，请稍后重试');
    }
}

/**
 * 打开淘宝购买
 */
function openTaobaoPurchase(url) {
    // WebView环境：通过JSBridge唤起淘宝App
    if (isInWebView()) {
        const bridges = [
            () => window.webkit?.messageHandlers?.openTaobao?.postMessage({ url }),
            () => window.AndroidInterface?.openTaobao?.(url),
            () => window.uni?.postMessage?.({ data: { action: 'openTaobao', url } })
        ];

        for (const bridge of bridges) {
            try {
                bridge();
                return;
            } catch (e) {
                console.warn('淘宝唤起失败:', e);
            }
        }
    }

    // 浏览器环境：直接打开链接
    window.open(url, '_blank');
}

/**
 * 打开支付页面
 */
function openPaymentPage(payUrl) {
    const bridges = [
        () => window.uni?.postMessage?.({ data: { action: 'openPaymentUrl', url: payUrl } }),
        () => window.uni?.webView?.postMessage?.({ data: { action: 'openPaymentUrl', url: payUrl } })
    ];

    for (const bridge of bridges) {
        try {
            bridge();
            return;
        } catch (e) {
            console.warn('打开支付页面失败:', e);
        }
    }

    // 回退方案
    window.location.href = payUrl;
}

/**
 * 检测是否在WebView环境
 */
function isInWebView() {
    const ua = navigator.userAgent.toLowerCase();
    return ua.includes('uni-app') ||
           ua.includes('wv') ||
           (typeof window.uni !== 'undefined') ||
           (typeof window.AndroidInterface !== 'undefined') ||
           (typeof window.webkit?.messageHandlers !== 'undefined');
}
